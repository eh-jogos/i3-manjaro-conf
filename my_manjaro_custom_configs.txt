        bindsym 1 exec terminal -hold -e $godot2; layout tabbed; mode "default"
        bindsym 2 exec terminal -hold -e $godot30; layout tabbed; mode "default"
        bindsym 3 exec terminal -hold -e $godot31; layout tabbed; mode "default"

        bindsym Return exec terminal -hold -e $godot2; layout tabbed; mode "default"
        bindsym Escape mode "default"
}
